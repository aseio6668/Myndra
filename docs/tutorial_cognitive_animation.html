Animation Grammar - Composable Movement Language

// Define movement as composable linguistic elements
animation_grammar = {
    // Basic movement "words"
    primitives: {
        "wiggle": |amplitude, frequency| sin_wave_motion(amplitude, frequency),
        "pulse": |intensity, rhythm| scale_animation(intensity, rhythm),
        "drift": |direction, chaos| procedural_movement(direction, chaos),
        "spiral": |center, radius| circular_motion(center, radius),
        "flock": |neighbors| boids_algorithm(neighbors)
    },
    
    // Movement "grammar" for complex compositions
    composition_operators: {
        // Sequential: one movement after another
        then: |motion1, motion2| chain_animations(motion1, motion2),
        
        // Parallel: multiple movements simultaneously  
        while: |motion1, motion2| blend_animations(motion1, motion2),
        
        // Conditional: movement depends on state
        if: |condition, motion| conditional_animation(condition, motion),
        
        // Probabilistic: maybe do this movement
        maybe: |probability, motion| random_animation(probability, motion),
        
        // Emotional: modify movement with feeling
        with_emotion: |motion, emotion| emotional_filter(motion, emotion)
    },
    
    // Complex "sentences" built from simple "words"
    examples: {
        curious_exploration: 
            "drift" with random_direction()
            then maybe(0.7, "wiggle" with small_amplitude())
            while "pulse" with heartbeat_rhythm()
            with_emotion("curiosity"),
            
        social_dance:
            "flock" with nearby_friends()

            with_emotion("joy"),
