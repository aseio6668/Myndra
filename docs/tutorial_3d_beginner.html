1. Your First 3D Scene - "Hello Cube"

Let's create your first 3D scene. In traditional 3D programming, this would take hundreds of lines. In Pomegranate3D, it's this simple:

// hello_cube_3d.pom
import space3d with capabilities("render")

fn main() {
    // Create a 3D world
    world {
        // Camera automatically positioned for good view
        camera { position: vec3(0, 0, 5) }
        
        // Add a simple cube
        mesh::cube {
            size: vec3(2, 2, 2)
            position: vec3(0, 0, 0)
            material: color(0.8, 0.2, 0.4)
        }
        
        // Basic lighting
        light::sun {
            direction: vec3(-1, -1, -1)
        }
    }
    
    // Start the 3D world
    render_world()
}
üéÆ What you see: A pink cube floating in 3D space with realistic lighting and shadows. The camera can be controlled with mouse/keyboard automatically.
ü™Ñ The Magic: Notice how we never mentioned vertices, shaders, render pipelines, or buffer management. Pomegranate3D handles all the complex 3D graphics programming under the hood!

2. Native 3D Data Types

Pomegranate3D includes built-in spatial types that make 3D programming intuitive:

// Built-in 3D types - no external math libraries needed!
let position = vec3(1.5, 2.0, -3.0)
let direction = vec3(0, 1, 0).normalized()
let rotation = quat::from_euler(45.degrees, 0.degrees, 90.degrees)

// Colors with alpha
let red = color(1.0, 0.0, 0.0, 1.0)
let transparent_blue = color(0.2, 0.4, 1.0, 0.7)

// 2D vectors for textures/UI
let texture_coord = vec2(0.5, 0.5)

// 4x4 matrices for advanced transforms
let transform_matrix = mat4::identity()
    .translate(vec3(2, 0, 0))
    .rotate_y(45.degrees)
    .scale(vec3(2, 1, 2))

// Automatic conversions between related types
let position_2d: vec2 = position.xy  // Gets x and y components
let bright_red = red * 1.5           // Scales color brightness

Vector Operations Made Simple

let a = vec3(1, 2, 3)
let b = vec3(4, 5, 6)

// Math operations work naturally
let sum = a + b                    // vec3(5, 7, 9)
let scaled = a * 2.0              // vec3(2, 4, 6)
let dot_product = a.dot(b)        // 32.0
let cross_product = a.cross(b)    // vec3(-3, 6, -3)
let distance = a.distance(b)      // 5.196
let direction = (b - a).normalized() // Unit vector from a to b

// Component access
let x = a.x      // 1.0
let xy = a.xy    // vec2(1, 2)
let length = a.magnitude()  // 3.742

3. Creating 3D Objects (Meshes)

Pomegranate3D provides built-in primitive shapes and easy custom mesh creation:

Basic Primitives

fn create_basic_shapes() {
    // Cube/Box
    mesh::cube {
        size: vec3(2, 1, 3)              // Width, height, depth
        position: vec3(-3, 0, 0)
        material: color(1.0, 0.5, 0.0)   // Orange
    }
    
    // Sphere
    mesh::sphere {
        radius: 1.5
        position: vec3(0, 0, 0)
        subdivisions: 32                  // Smoothness level
        material: color(0.2, 0.8, 0.2)   // Green
    }
    
    // Cylinder
    mesh::cylinder {
        radius: 1.0
        height: 3.0
        position: vec3(3, 0, 0)
        material: color(0.2, 0.4, 1.0)   // Blue
    }
    
    // Plane (great for ground/walls)
    mesh::plane {
        size: vec2(10, 10)
        position: vec3(0, -2, 0)
        material: color(0.7, 0.7, 0.7)   // Gray ground
    }
}

Custom Mesh from Vertices

// Create a triangle
mesh::custom {
    vertices: [
        vec3(0, 1, 0),      // Top vertex
        vec3(-1, -1, 0),    // Bottom left
        vec3(1, -1, 0)      // Bottom right
    ]
    
    // Vertex colors (optional)
    colors: [
        color(1, 0, 0),     // Red top
        color(0, 1, 0),     // Green bottom left  
        color(0, 0, 1)      // Blue bottom right
    ]
    
    // Triangles (vertex indices)
    triangles: [
        [0, 1, 2]           // Connect vertices 0, 1, 2
    ]
    
    position: vec3(0, 0, 0)
}

4. Materials and Appearance

Make your 3D objects look amazing with built-in material system:

// Simple colored materials
material: color(0.8, 0.2, 0.6)

// Physically-Based Rendering (PBR) materials
material: materials::pbr {
    albedo: color(0.7, 0.1, 0.1)     // Base color
    metallic: 0.8                     // How metallic (0.0 to 1.0)
    roughness: 0.2                    // Surface roughness (0.0 = mirror)
    emissive: color(0.1, 0.0, 0.0)    // Glow color
}

// Glass/transparent materials
material: materials::glass {
    color: color(0.9, 0.9, 1.0, 0.3)  // Tinted blue glass
    refraction_index: 1.5              // How light bends through it
}

// Textured materials
material: materials::textured {
    diffuse_texture: "brick.jpg"       // Base color texture
    normal_texture: "brick_normal.jpg" // Surface detail bumps
    roughness_texture: "brick_rough.jpg" // Roughness variation
}

// Animated/changing materials
let dynamic_cube = mesh::cube {
    size: vec3(1, 1, 1)
    material: materials::pbr {
        albedo: color(1.0, 0.5, 0.5)
        emissive: color(sin(time), cos(time), 0.5)  // Animated glow!
    }
}

5. Cameras and Viewing

Control how you see your 3D world:

// Perspective camera (most common)
camera perspective_cam {
    position: vec3(0, 5, 10)
    target: vec3(0, 0, 0)              // What to look at
    fov: 60.degrees                    // Field of view
    near: 0.1                          // Near clipping plane
    far: 1000.0                        // Far clipping plane
}

// Orthographic camera (no perspective distortion)
camera ortho_cam {
    position: vec3(10, 10, 10)
    target: vec3(0, 0, 0)
    size: 20.0                         // How much area to show
    near: 0.1
    far: 100.0
}

// First-person camera with automatic controls
camera first_person {
    position: vec3(0, 1.7, 0)          // Eye height
    movement_speed: 5.0                // Units per second
    mouse_sensitivity: 2.0
    enable_wasd: true                  // WASD movement
    enable_mouse_look: true            // Mouse camera control
}

6. Lighting Your Scene

Add realistic lighting with simple declarations:

lighting {
    // Ambient light - soft light everywhere
    ambient: color(0.2, 0.2, 0.3)
    
    // Directional light (like the sun)
    light::directional {
        direction: vec3(-1, -2, -1).normalized()
        color: color(1.0, 0.9, 0.8)     // Warm sunlight
        intensity: 2.0
        cast_shadows: true
    }
    
    // Point light (like a lightbulb)
    light::point {
        position: vec3(3, 4, 2)
        color: color(1.0, 0.8, 0.6)     // Warm white
        intensity: 5.0
        range: 10.0                      // How far light reaches
        cast_shadows: true
    }
    
    // Spot light (like a flashlight)
    light::spot {
        position: vec3(0, 5, 0)
        direction: vec3(0, -1, 0)        // Pointing down
        color: color(0.8, 1.0, 0.8)     // Cool light
        intensity: 3.0
        inner_angle: 30.degrees          // Bright center
        outer_angle: 45.degrees          // Fade to dark
        range: 15.0
    }
}

7. Making Things Move

Animate your 3D objects using Pomegranate's temporal types:

fn create_animated_scene() {
    let spinning_cube = mesh::cube {
        size: vec3(1, 1, 1)
        position: vec3(0, 0, 0)
        material: color(1.0, 0.3, 0.7)
    }
    
    // Continuous rotation
    spinning_cube.rotation.y -> spinning_cube.rotation.y + 360.degrees over 3s then loop
    
    let bouncing_sphere = mesh::sphere {
        radius: 0.8
        position: vec3(3, 2, 0)
        material: materials::pbr {
            albedo: color(0.2, 0.8, 0.2)
            metallic: 0.1
        }
    }
    
    // Bouncing animation
    bouncing_sphere.position.y -> 4.0 over 1s 
        then bouncing_sphere.position.y -> 0.5 over 1s 
        then loop
    
    // Scale animation
    bouncing_sphere.scale -> vec3(1.2, 1.2, 1.2) over 0.5s
        then bouncing_sphere.scale -> vec3(1.0, 1.0, 1.0) over 0.5s
        then loop
}

9. Complete Example: Floating Island

import space3d with capabilities("render", "physics")

fn main() {
    world {
        // Nice camera angle
        camera {
            position: vec3(8, 6, 12)
            target: vec3(0, 0, 0)
            fov: 45.degrees
        }
        
        // Atmospheric lighting
        lighting {
            ambient: color(0.3, 0.4, 0.6)
            
            light::directional {
                direction: vec3(-0.5, -1.0, -0.3).normalized()
                color: color(1.0, 0.9, 0.7)
                intensity: 1.8
                cast_shadows: true
            }
        }
    }
    
    create_floating_island()
    render_world()
}

fn create_floating_island() {
    // Island base
    mesh::cylinder {
        radius: 4.0
        height: 1.0
        position: vec3(0, 0, 0)
        material: materials::pbr {
            albedo: color(0.4, 0.3, 0.2)  // Brown earth
            roughness: 0.9
        }
    }
    
    // Grass on top
    mesh::cylinder {
        radius: 4.2
        height: 0.1
        position: vec3(0, 0.55, 0)
        material: color(0.2, 0.8, 0.3)   // Green grass
    }
    
    // Tree trunk
    mesh::cylinder {
        radius: 0.3
        height: 3.0
        position: vec3(1.5, 2.0, 0)
        material: color(0.4, 0.2, 0.1)   // Brown trunk
    }
    
    // Tree leaves
    mesh::sphere {
        radius: 1.5
        position: vec3(1.5, 4.0, 0)
        material: color(0.1, 0.6, 0.1)   // Dark green leaves
    }
    
    // Floating crystals around the island
    for i in 0..6 {
        let angle = i * 60.degrees
        let crystal = mesh::icosphere {
            radius: 0.3
            position: vec3(
                sin(angle) * 6.0,
                2.0 + sin(angle * 2.0) * 0.5,
                cos(angle) * 6.0
            )
            material: materials::glass {
                color: color(0.8, 0.9, 1.0, 0.8)
                emissive: color(0.2, 0.3, 0.8) * sin(time + i)
            }
        }
        
        // Gentle floating animation
        crystal.position.y -> crystal.position.y + 0.5 over 2s
            then crystal.position.y -> crystal.position.y - 0.5 over 2s
            then loop
        
        // Slow rotation
        crystal.rotation -> crystal.rotation + vec3(0, 180.degrees, 0) over 4s then loop
    }
}
